Sub MyIp()
Dim s1 As String
Dim s2 As String
Dim s1_ As String
Dim s2_ As String
Dim ip As String
Dim num_ip As String
For i = 2 To 40
    Worksheets("4этаж").Activate
    s1 = "E" + CStr(i)
    num_ip = "B" + CStr(i)
    ip = Range(num_ip)
    s1_ = Range(s1)
    For j = 2 To 135
        Worksheets("Список").Activate
        s2 = "d" + CStr(j)
        s2_ = Range(s2)
            If (s1_ = s2_) Then
                s2 = "G" + CStr(j)
                Range(s2) = ip
            End If
    Next j
 Next i
End Sub
